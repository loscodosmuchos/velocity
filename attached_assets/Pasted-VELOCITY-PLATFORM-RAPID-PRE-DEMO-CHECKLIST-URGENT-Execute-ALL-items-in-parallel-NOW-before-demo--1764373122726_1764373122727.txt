CRITICAL ERROR PATTERNS (Fix Immediately)
ERROR 1: "Cannot read property X of undefined"
Symptom: React component crashes when accessing data
Root Cause: Missing null checks on async data
Fix Pattern:

typescript
// BAD
const { data } = useQuery('sows');
return <div>{data.name}</div>; // CRASH if data undefined

// GOOD
const { data, isLoading } = useQuery('sows');
if (isLoading) return <Skeleton />;
if (!data) return <div>No data</div>;
return <div>{data.name}</div>;

#!/usr/bin/env node
const { chromium } = require('playwright');

async function preDemoCheck() {
  console.log('üîç RAPID PRE-DEMO CHECK\n');
  
  const browser = await chromium.launch();
  const page = await browser.newPage();
  
  const checks = { passed: [], failed: [] };
  const url = process.env.DEMO_URL || 'http://localhost:3000';
  
  try {
    // Check 1: Homepage loads
    await page.goto(url, { timeout: 5000 });
    await page.waitForSelector('.command-center', { timeout: 5000 });
    checks.passed.push('‚úÖ Homepage loads (<5s)');
    
    // Check 2: Document Analyzer accessible
    const analyzerBtn = page.locator('[data-test="document-analyzer"]');
    const isClickable = await analyzerBtn.isEnabled();
    if (isClickable) {
      checks.passed.push('‚úÖ Document Analyzer button clickable');
    } else {
      checks.failed.push('‚ùå Document Analyzer button DISABLED');
    }
    
    // Check 3: Demo mode doesn't break functionality
    await page.goto(url + '?mode=demo');
    const demoBtnClickable = await analyzerBtn.isEnabled();
    if (demoBtnClickable) {
      checks.passed.push('‚úÖ Demo mode preserves button functionality');
    } else {
      checks.failed.push('‚ùå Demo mode BREAKS buttons');
    }
    
    // Check 4: API health
    const response = await page.goto(url + '/api/health');
    if (response.ok()) {
      checks.passed.push('‚úÖ API health endpoint responds');
    } else {
      checks.failed.push('‚ùå API health check FAILED');
    }
    
  } catch (error) {
    checks.failed.push(`‚ùå CRITICAL: ${error.message}`);
  } finally {
    await browser.close();
  }
  
  // Report
  console.log('‚ïê'.repeat(60));
  console.log(`PASSED (${checks.passed.length}):`);
  checks.passed.forEach(c => console.log(`  ${c}`));
  
  if (checks.failed.length > 0) {
    console.log(`\nFAILED (${checks.failed.length}):`);
    checks.failed.forEach(c => console.log(`  ${c}`));
    console.log('\nüö® DEMO NOT READY - FIX ISSUES BEFORE PROCEEDING\n');
    process.exit(1);
  } else {
    console.log('\n‚úÖ ALL CHECKS PASSED - DEMO READY\n');
    process.exit(0);
  }
}
